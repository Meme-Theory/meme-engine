# ---- INSTALL ----------------------------------------------------------------
#
#   # Replace {{COMPUTE_VENV_PY}} with the venv interpreter path, e.g.:
#   #   Windows:  {{COMPUTATION_DIR}}/.venv/Scripts/python.exe

#
#   "{{COMPUTE_VENV_PY}}" -m pip install -r requirements-compute.txt
#
# ---- WHERE USED -------------------------------------------------------------
#
#   {{COMPUTATION_DIR}}/*.py   - all active compute scripts
#   (discipline packs may add further script paths in their overlay notes)
#
# ---- NOT FOR ----------------------------------------------------------------
#
#   tools/mcp-servers/ - use requirements-mcp.txt installed into system Python
#   3.13. MCP servers run as stdio subprocesses spawned by Claude Code; they do
#   not need torch, cvxpy, camb, or anything GPU-related.
#
# ---- ENV VARS (set BEFORE numpy import in CPU-bound scripts) ---------------
#
#   import os
#   os.environ.setdefault('OMP_NUM_THREADS', '8')
#   os.environ.setdefault('MKL_NUM_THREADS', '8')
#   # Prevents numpy/scipy from fanning out to all cores when multiple
#   # compute-mode agents run in parallel. Tune to your core count.

# ==============================================================================
#  GPU (ROCm / CUDA) - install MANUALLY, NOT via pip install -r
# ==============================================================================
#
# The ROCm / CUDA torch wheel MUST match your locally installed GPU SDK, and
# the wheel lives on PyTorch's ROCm or CUDA index, not PyPI. Pinning it here in
# a plain-index requirements file would break reproducibility across machines.
# Install separately, using the command appropriate for your hardware:
#
#   # AMD ROCm (example - match to your ROCm SDK version):
#   "{{COMPUTE_VENV_PY}}" -m pip install \
#       torch==2.9.1+rocmsdk20260116 \
#       --index-url https://download.pytorch.org/whl/rocm7.2
#
#   # NVIDIA CUDA (example):
#   "{{COMPUTE_VENV_PY}}" -m pip install \
#       torch \
#       --index-url https://download.pytorch.org/whl/cu124
#
#   # CPU-only torch (if no GPU):
#   "{{COMPUTE_VENV_PY}}" -m pip install torch
#
# Used by:
#   - torch.linalg.eigvals / .svd / .matmul on GPU for large matrices
#   - torch.fft for large FFTs
#   - First-use validation: cross-check first few eigenvalues against numpy.linalg
#     on a small test matrix before trusting GPU results.
